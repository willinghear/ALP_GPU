# ALP_src/alp_lib/CMakeLists.txt
add_library(ALP SHARED
    src/falp.cpp
    src/fastlanes_ffor.cpp
    src/fastlanes_generated_ffor.cpp
    src/fastlanes_generated_unffor.cpp
    src/fastlanes_unffor.cpp
)

# 设置 include 目录
target_include_directories(ALP PUBLIC include)

# 可选的 Clang-Tidy 设置
find_program(CLANG_TIDY_EXE NAMES clang-tidy)
if (CLANG_TIDY_EXE)
    set_target_properties(ALP PROPERTIES CXX_CLANG_TIDY "${CLANG_TIDY_EXE}")
endif ()
